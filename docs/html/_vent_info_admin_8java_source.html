188 public List<ReservaData> obtenerReservas(String fecha, int id) throws Exception {
189
190
191 List<ReservaData> listData = getReservas(id);
192 List<ReservaData> reservasFecha = new ArrayList<>();
193 Date da1 = new SimpleDateFormat("dd/MM/yyyy").parse(fecha);
194 for (ReservaData reserva: listData) {
195 Date dateFechaI = new SimpleDateFormat("yyyy-MM-dd").parse(reserva.getFecha_ini());
196 Date dateFechaF = new SimpleDateFormat("yyyy-MM-dd").parse(reserva.getFecha_fin());
197 if((da1.equals(dateFechaI) || da1.after(dateFechaI)) && da1.before(dateFechaF)) {
198 reservasFecha.add(reserva);
199 }
200 }
